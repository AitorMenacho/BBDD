CREATE OR REPLACE FUNCTION SALARIO_TOTAL
(V_EMPNO EMP.EMPNO%TYPE)
RETURN NUMBER
AS

   V_SUMA NUMBER(10) := 0;

BEGIN

    SELECT SUM(EMP.SAL + EMP.COMM) INTO V_SUMA
    FROM EMP
    WHERE EMPNO = V_EMPNO;
    
    RETURN V_SUMA;

END;
